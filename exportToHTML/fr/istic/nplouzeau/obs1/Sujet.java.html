8    import java.lang.IllegalArgumentException; 
9     
10   public interface Sujet<T> { 
11       /** 
12        * Ajoute un observateur o, à condition que o ne soit pas déjà abonné à this 
13        * 
14        * @param o L'observateur à ajouter 
15        * @throws IllegalArgumentException si o est null ou o est déjà abonné à this 
16        */ 
17       void abonner(Observateur<T> o) throws IllegalArgumentException; 
18    
19       /** 
20        * Retire un observateur o de la liste des abonnés à this, 
21        * à condition que 
22        * o soit abonné 
23        * 
24        * @param o L'observateur à retirer 
25        * @throws IllegalArgumentException si o est null ou o n'est pas abonné à this 
26        */ 
27       void désabonner(Observateur<T> o) throws IllegalArgumentException; 
28    
29       /** 
30        * Lit la valeur du sujet 
31        * 
32        * @return valeur du sujet 
33        */ 
34       T getValeur(); 
35    
36       /** 
37        * Indique si un observateur est abonné au sujet this 
38        * 
39        * @param o l'observateur à tester 
40        * @return true ssi o est déjà abonné 
41        * @throws IllegalArgumentException si o est null 
42        */ 
43       boolean estAbonné(Observateur<T> o); 
44    
45       /** 
46        * Change la valeur du sujet 
47        * 
48        * @param v la nouvelle valeur du sujet 
49        */ 
50       void setValeur(T v); 
51    
52   }
